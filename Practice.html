Program to generate Fibonnaci series upto n numbers

A Fibonnaci series is a series in which every term in the series is the sum of previous two terms.
For example, 1 1 2 3 5 8.....

#include<iostream>
using namespace std;
int main()
{
int n,x=0,y=1,nextterm;
cout<"Enter a number: ";
cin>>n;
cout<"Series: ";
for(int i=0;i<=n;i++)
{
cout<x<" ";
nextterm=x+y;
x=y;
y=nextterm;
}
return 0;
}

Program to show function overloading

You have to create more than one function with same name and with different arguments.Create as many functions as you want, then try to call each from the main function.

#include<iostream>
using namespace std;
class A
{
public:
void sum(int x,int y) //overloaded function
{ cout<<"Sum is "<<x+y;}
void sum(double i,double j) //overloaded function
{ cout<<"Sum is:"<<i+j;}
void sum(double i,double j,double k) //overloaded function
{ cout<<"Sum is:"<<i+j+k;}
};
int main()
{
A a1;

a1.sum(11.5,2.5,1.5);

Program to demonstrate the overloading of * operator

You have to create a class where you have to define a method to overload * operator.

#include<iostream>
using namespace std;
class A
{
int a,b;
public:
A(){ a = b = 0;}
A(int x,int y)
{a = x; b = y;}
void display()
{cout<<"a:"<<a<<" b:"<<b;}
A operator* (int i) //operator overloading
{
A temp;
temp.a = i*a;
temp.b = i*b;
return(temp);
}
};
int main()
{
A a1(3,4);
A a2;
a2 = a1*2;
a2.display();
return 0;
}

Program to demonstrate the overloading of increment and decrement operator.

Similar as overloading of other operators. Just a slight difference in overloading of increment ans decrement operator.

#include<iostream>
using namespace std;
class A
{
int a;
public:
void setData(int x)
{ a = x;}
void display()
{cout<<"a:"<<a<<endl;}
A operator++(int)
{
A temp;
temp.a = a++;
return(temp);
}
A operator--(int)
{
A temp;
temp.a = a--;
return(temp);
}
};
int main()
{
A a1,a2,a3;
a1.setData(10);
a2 = a1++;
a2.display();
a3 = a1--;
a3.display();
return 0;
}
