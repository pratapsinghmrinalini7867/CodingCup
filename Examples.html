Examples

Copy Constructors

#include<iostream>
using namespace std;
class A
{
int a,b;
public:
A(int x, int y) //copy constructor
{a = x; b = y;}
A(A &x)
{ a = x.a; b = x.b; }
void display()
{cout<<"a:"<<a<<" b:<<"b;}
};
int main()
{
A a1(3,4);
A a2(a1);
a1.display();
a2.display();
return 0;
}

MultiLevel Inheritance

#include<iostream>
using namespace std;
class A
{
public:
int a;
A()
{ a = 10; }
};
class B:private A
{
public:
int b;
B()
{ b = 5;}
int get()
{ return a;}
void sum()
{ cout<<a+b;}
};
class C: public B
{
public:
int c;
C()
{ c = 20;}
};

int main()
{
B b1;
C c1;
b1.sum();
cout<<"\n"<<c1.a<<" "<<c1.b<<" "<<c1.c<<endl;
return 0;
}

Function Overloading

#include<iostream>
using namespace std;
class A
{
public:
void sum(int x,int y) //overloaded function
{ cout<<"Sum is "<<x+y;}
void sum(double i,double j) //overloaded function
{ cout<<"Sum is:"<<i+j;}
void sum(double i,double j,double k) //overloaded function
{ cout<<"Sum is:"<<i+j+k;}
};
int main()
{
A a1;
a1.sum(3,4);
a1.sum(12.5,2.5);
a1.sum(11.5,2.5,1.5);
return 0;
}

Operator Overloading

#include<iostream>
using namespace std;
class A
{
int a,b;
public:
A(){ a = b = 0;}
A(int x,int y)
{a = x; b = y;}
void display()
{cout<<"a:"<<a<<" b:"<<b;}
A operator+ (A x) //operator overloading
{
A temp;
temp.a = a + x.a;
temp.b = b + x.b;
return(temp);
}
};
int main()
{
A a1(3,4),a2(5,6);
A a3;
a3 = a1+a2;
a3.display();
return 0;
}

Overloading of Incement and Decrement opertaor

#include<iostream>
using namespace std;
class A
{
int a;
public:
void setData(int x)
{ a = x;}
void display()
{cout<<"a:"<<a<<endl;}
A operator++(int)
{
A temp;
temp.a = a++;
return(temp);
}
A operator--(int)
{
A temp;
temp.a = a--;
return(temp);
}
};
int main()
{
A a1,a2,a3;
a1.setData(10);
a2 = a1++;
a2.display();
a3 = a1--;
a3.display();
return 0;
}

Type Conversion

Basic type to Class type

#include<iostream>
using namespace std;
class square
{
int a;
public:
void setData(int x)
{ a=x;}
square(){}
//this method will convert int type to square type
square(int z)
{
a=z;
}
void display()
{cout<<"a:"<<a<<endl;}
};
int main()
{
square s;
int x = 10;
s = x;
s.display();
return 0;
}

Class type to Base type

#include<iostream>
using namespace std;
class square
{
int a;
public:
void setData(int x)
{ a=x;}
int getValue()
{ return(a);}
//Class Type to Primitive Type
operator int()
{
return(a);
}
void display()
{cout<<"a:"<<a<<endl;}
};
int main()
{
square s;
s.setData(10);
int x;
x = s;
cout<<"x:"<<x<<endl;
s = x;
s.display();
return 0;
}
